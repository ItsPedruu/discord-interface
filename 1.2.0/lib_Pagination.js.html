    const EventEmitter = require('events').EventEmitter;
const Constants = require('../utils/Constants.js');

/**
 * Represents a Pagination Interface
 * @class
 */
class Pagination extends EventEmitter {
    /**
     * @param {Message} message The discord.js message object
     * @param {PaginationOptions} options Options for the Pagination Interface
     */
    constructor(message, options) {
        super();

        if (!message) throw new SyntaxError('Expected message');
        if (!options) throw new SyntaxError('Expected options');

        if (!options.userID || !options.getPage || !options.items) throw new SyntaxError('Expected options.userID and options.getPage and options.items');

        this.message = message;
        this.options = Object.assign(Constants.PaginationOptions, options);

        this.start();
    }

    async start() {
        let current = 0;
        let message = await this.message.channel.send(this.options.getPage(this.options.items[current], current + 1));

        let reactions = this.options.items.length > 2 ? [...Object.values(this.options.reactions), ...(this.options.extraReactions || [])] : this.options.items.length == 1 ? this.options.extraReactions : [this.options.reactions.previousPage, ...(this.options.extraReactions || []), this.options.reactions.nextPage, this.options.reactions.cancel];

        let filter = (reaction, user) => (reactions.includes(reaction.emoji.name) || reactions.includes(reaction.emoji.id)) && user.id == this.options.userID;
        this.collector = message.createReactionCollector(filter, this.options);

        this.collector.on('collect', async reaction => {
            if (reaction.emoji.name == this.options.reactions.cancel) return this.collector.stop();
            
            if (this.options.removeReactions) await reaction.remove(message.author.id);

            if (reaction.emoji.name == this.options.reactions.nextPage) {
                current = (current + 1) % this.options.items.length;
            } else if (reaction.emoji.name == this.options.reactions.previousPage) {
                current = current - 1 < 0 ? this.options.items.length - 1 : current - 1;
            } else if (reaction.emoji.name == this.options.reactions.backToStart) {
                current = 0;
            } else if (reaction.emoji.name == this.options.reactions.skipToFinal) {
                current = this.options.items.length - 1;
            } else if (this.options.extraReactions.includes(reaction.emoji.name) || this.options.extraReactions.includes(reaction.emoji.id)) {
                let extraReactionCollected = this.options.extraReactions.includes(reaction.emoji.name) ? reaction.emoji.name : reaction.emoji.id;

                this.emit('collect', extraReactionCollected);
            }

            message.edit(this.options.getPage(this.options.items[current], current + 1));
        });

        this.collector.on('end', async (collected, reason) => {
            if (!message.deleted) await message.delete();
            this.emit('over', reason == 'time');
        })

        for (let reaction of reactions) {
            if (message.deleted) return this.collector.stop();
            await message.react(reaction);
        }   
    }

    /**
     * Creates a Pagination
     * @param {Message} message The discord.js message object
     * @param {PaginationOptions} options The options for the Pagination Interface
     */
    static create(message, options) {
        return new Pagination(message, options);
    }
}

 * @event Pagination#over
 * @property {boolean} timeOver Whether the reason the collector stopped is due to the time over
 */

module.exports = Pagination;
